number_of_steppe_provinces_in_realm = {
	value = 0
	every_sub_realm_county = {
		every_county_province = {
			if = {
				limit = {
					OR = {
						terrain = steppe
						terrain = plains
						terrain = desert
					}
				}
				add = 1
			}
			else = { add = 0 }
		}
	}
}
number_of_realm_provinces = {
	value = 0
	every_sub_realm_county = { every_county_province = { add = 1 } }
}
half_number_of_realm_provinces = {
	value = 0
	every_sub_realm_county = { every_county_province = { add = 1 } }
	divide = 2
}
